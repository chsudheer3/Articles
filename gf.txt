n=["add","spliter","my_max"]
a=[1,2,3]
def add(a):
    sum=0
    for i in a:
        sum=sum+i
    return sum
    
def square(a):
    return a*a
    
def spliter(a):
    f=a//2
    return [f, a-f]

def my_max(a):
    if isinstance(a, (list,tuple)):
      return max(a)
    return a
    
def my_min(a):
    if isinstance(a, (list,tuple)):
      return min(a)
    return a
    
maps={"add":add,  "square":square, "spliter":spliter, "my_max":my_max, "my_min":my_min}
for i in n:
    result = maps[i](a)
    a=result
print(a)
